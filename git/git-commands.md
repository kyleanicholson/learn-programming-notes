```shell
git status # shows files that are staged/unstaged
git remote -v # view remote repository
git commit -m "<message>" #creates a commit with comment 
git log # shows recent commits
git add <filename> # add file to staging for commit
	git add . # adds all files in current directory to staging
git push or git push origin main # push to remote repo (main is branch name)

# clone remote repo, recommended to use SSH
git clone git@github.com:USER-NAME/REPOSITORY-NAME.git 

# rebase



```
